    public abstract class AbstractMapMsgListener implements MessageListener<String, byte[]> {

    /** 是否有异常标识: 未知表或者数据查询失败等, 不执行handleSuccess以及不输出增量成功日志 */
    protected boolean notError;
    /** 操作类别insert(0),delete(1), update(2) */
    protected Obj.OptType operate;

    @Override
    public Action consume(ConsumerRecord<String, byte[]> record, ConsumeContext context) {
        long startTime = System.currentTimeMillis();
        ZorroEvent zorroEvent;
        try {
            zorroEvent = ZorroEvent.parseFrom(record.value());
        } catch (InvalidProtocolBufferException e) {
            log.error("kafka consumer 解析错误", e);
            return Action.DiscardInvalidMessage;
        }

        if (zorroEvent != null) {
            notError = true;
            operate = zorroEvent.getOptType();

            Map<String, String> data;
            if(Obj.OptType.delete == operate) {
                data = zorroEvent.getBeforeMap();
            } else {
                data = zorroEvent.getAfterMap();
            }

            String id = data.get(SearchConstants.EntityProperties.ID);
            String db = zorroEvent.getDb();
            String table = zorroEvent.getTb();
            boolean isSuccess;
            if (Obj.OptType.insert == operate) {
                isSuccess = insert(db, table, data);
            }
            else if (Obj.OptType.update == operate) {
                isSuccess = update(db, table, data);
            }
            else if (Obj.OptType.delete == operate) {
                isSuccess = delete(db, table, data);
            }
            else {
                String errMsg = buildLogMsg("unknown operate", db, table, id, operate);
                log.info(errMsg);
                return Action.CommitMessage;
            }

            if(!isSuccess) {
                String errMsg = buildLogMsg("增量请求失败", db, table, id, operate);
                log.warn(errMsg + ", key={}, topic={}, partition={}, offset={}", record.key(), record.topic(),
                        record.partition(), record.offset());
                handleFailed(db, table, data);
                return Action.ReconsumeLater;
            }

            if(notError) {
                handleSuccess(db, table, data);
                String logMsg = buildLogMsg("增量请求成功", db, table, id, operate);
                log.info(logMsg + ", 耗时{}ms", System.currentTimeMillis() - startTime);
            }
        }
        return Action.CommitMessage;
    }

    @Slf4j
@Component
public class CmdsRegionCache {

    private static final List<DictItem> regionCache = new CopyOnWriteArrayList<>();

    private final ScheduledExecutorService executor = Executors.newSingleThreadScheduledExecutor();

    /** cmds查询服务 */
    private final DictService dictService;

    @Autowired
    public CmdsRegionCache(DictService dictService) {
        this.dictService = dictService;
    }

    private List<DictItem> load() throws CacheException {
        if(regionCache.isEmpty()) {
            synchronized (this) {
                if(regionCache.isEmpty()) {
                    List<DictItem> dictItems;
                    try {
                        dictItems = getDictItems();
                    } catch (MdsException e) {
                        throw new CacheException(e);
                    }

                    regionCache.addAll(dictItems);
                    expire(24, TimeUnit.HOURS);
                    log.info("投保地缓存完成.");
                }
            }
        }
        return regionCache;
    }

    /**
     * 指定缓存过期时间
     *
     * @param time 如果小于1, 立即清空
     * @param unit {@link TimeUnit}
     */
    public void expire(long time, TimeUnit unit) {
        if(time > 0) {
            executor.schedule(()-> {
                regionCache.clear();
                log.info("投保地缓存定时清理结束.");
            }, time, unit);
        } else {
            regionCache.clear();
            log.info("投保地缓存清理结束.");
        }
    }

    /**
     * 获取所有地区数据
     *
     * @return {@link DictItem}
     */
    public List<DictItem> get() throws CacheException {
        List<DictItem> cache = load();
        List<DictItem> items = new ArrayList<>();
        items.addAll(cache);
        return items;
    }

    /**
     * 获取投保地信息
     *
     * @param insurePlaceCode 投保地code
     * @return {@link DictItem} or null
     */
    public DictItem get(String insurePlaceCode) {
        try {
            List<DictItem> cache = load();
            for (DictItem region : cache) {
                if(StringUtils.equals(region.getCode(), insurePlaceCode)) {
                    return region;
                }
            }
        } catch (CacheException e) {
            log.error(e.getMessage(), e);
        }
        return null;
    }

    /** 获取所有地区数据. 查询主数据71 */
    public List<DictItem> getDictItems() throws MdsException {
        ResultBase<Dict> result = dictService.loadDictByCode(CmdsUtil.SYSTEM_CODE, CmdsUtil.DICT_CODE, CmdsUtil.VERSION);
        if (result.isSuccess()) {
            List<DictItem> items = result.getValue().getItems();
            return items;
        } else {
            throw new MdsException(CmdsErrorCode.LOAD_UNSUCCESS, JSON.toJSONString(result));
        }
    }

}
